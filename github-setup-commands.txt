# GitHub Setup Commands
# Replace 'yourusername' with your actual GitHub username

# Add the remote repository (replace yourusername with your GitHub username)
git remote add origin https://github.com/yourusername/texna-website.git

# Push to GitHub
git branch -M main
git push -u origin main

# Alternative if you prefer SSH (make sure you have SSH keys set up)
# git remote add origin git@github.com:yourusername/texna-website.git
# git push -u origin main

# To verify the remote was added correctly
git remote -v

# To check the status
git status
